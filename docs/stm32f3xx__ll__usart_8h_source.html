1613{

1615 uint32_t brrresult = 0x0U;
1616
1617 usartdiv = USARTx->BRR;
1618
1619 if (usartdiv == 0U)
1620 {
1621 /* Do not perform a division by 0 */

1630 }
1631 else
1632 {
